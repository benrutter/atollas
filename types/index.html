def datetime_tz(tz=str):
    """
    Return datetime_tz type for a given timezone.
    Will re represented as a numpy type, for intance
    if "UTC" is given, then "datetime64[ns, UTC}"
    will be used.
    """
    return ColumnType(f"datetime64[ns, {tz}]")
